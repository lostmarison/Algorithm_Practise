# 110.平衡二叉树
```c++
class Solution {
public:
    int height(TreeNode* bt) {
        if (!bt) {
            return 0;
        }
        return max(height(bt->left), height(bt->right)) + 1;
    }
    bool isBalanced(TreeNode* root) {
        if (!root) {
            return true;
        } else {
            return abs(height(root->left) - height(root->right)) <= 1 &&
                   isBalanced(root->left) && isBalanced(root->right);
        }
    }
};
```
