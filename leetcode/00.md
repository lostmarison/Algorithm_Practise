class Solution {
    void ovenevenList(Node* head) {
        if (!head || !head->next || !head->next->next) { // 至少有3个结点
            return head;
        }
        Node* odd = head;
        Node* even = head->next;
        Node* p1 = odd;
        Node* p2 = even;
        while (p2->next && p2->next->next) {
            p1->next = p2->next;
            p1 = p1->next;
            p2->next = p1->next;
            p2 = p2->next;
        }
        if (p2->next) {
            p1->next = p2->next;
            p1 = p1->next;
        }
        p1->next = even;
        p2->next = nullptr;

        return odd;
    }
};
