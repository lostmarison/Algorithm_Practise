# 617.合并二叉树
[leetcode617](https://leetcode.cn/problems/merge-two-binary-trees/)
```c++
class Solution {
public:
    /**
     * 合并两棵二叉树
     *
     * @param root1 第一棵二叉树的根节点
     * @param root2 第二棵二叉树的根节点
     * @return 合并后的二叉树的根节点
     */
    TreeNode* mergeTrees(TreeNode* root1, TreeNode* root2) {
        // 如果root1为空，则返回root2作为合并后的树
        if (!root1) {
            return root2;
        }
        // 如果root2为空，则返回root1作为合并后的树
        if (!root2) {
            return root1;
        }

        // 创建新节点，其值为root1和root2节点值之和
        TreeNode* root = new TreeNode(root1->val + root2->val);

        // 递归地合并左子树，并将合并后的子树连接到新节点的左子节点上
        root->left = mergeTrees(root1->left, root2->left);

        // 递归地合并右子树，并将合并后的子树连接到新节点的右子节点上
        root->right = mergeTrees(root1->right, root2->right);

        // 返回合并后的树的根节点
        return root;
    }
};
```
