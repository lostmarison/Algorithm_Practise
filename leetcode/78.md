```c++
/*
void backtracking(参数) {
    if (终止条件) {
        存放结果;
        return;
    }

    for (选择：本层集合中元素（树中节点孩子的数量就是集合的大小）) {
        处理节点;
        backtracking(路径，选择列表); // 递归
        回溯，撤销处理结果
    }
}
*/
class Solution {
private:
    vector<vector<int>> result;
    vector<int> path;

    void backtracking(vector<int>& nums, int startIndex) {
        result.push_back(path);
        if (startIndex >= nums.size()) {
            return;
        }
        for (int i = startIndex; i < nums.size(); ++i) {
            path.push_back(nums[i]);
            backtracking(nums, i + 1);
            path.pop_back();
        }
    }

public:
    vector<vector<int>> subsets(vector<int>& nums) {
        result.clear();
        path.clear();
        backtracking(nums, 0);
        return result;
    }
};
```

```c++
/*
初始调用:
    subsets(nums) 被调用，result 和 path 被清空。调用 backtracking(nums, 0)。

1.第一次递归 (startIndex = 0):
    当前路径 path 为 []，将其添加到 result。
    进入循环，i = 0：
        选择 1，path 变为 [1]，调用 backtracking(nums, 1)。

2.第二次递归 (startIndex = 1):
    当前路径 path 为 [1]，添加到 result。
    进入循环，i = 1：
        选择 2，path 变为 [1, 2]，调用 backtracking(nums, 2)。

3.第三次递归 (startIndex = 2):
    当前路径 path 为 [1, 2]，添加到 result。
    进入循环，i = 2：
        选择 3，path 变为 [1, 2, 3]，调用 backtracking(nums, 3)。

4.第四次递归 (startIndex = 3):
    当前路径 path 为 [1, 2, 3]，添加到 result。
    由于 startIndex 超出 nums 的大小，返回到上层。

5.回溯：
    撤销选择，path 变回 [1, 2]。
    继续循环，i 增加到 3，没有更多选择，返回到上层。

6.继续第二次递归：
    撤销选择，path 变回 [1]。
    继续循环，i = 2：
        选择 3，path 变为 [1, 3]，调用 backtracking(nums, 3)。


7.重复到第四次递归:
    当前路径 [1, 3] 添加到 result，返回。

8.继续回溯：
    撤销选择，path 变回 [1]，返回到第一次递归。

9.继续第一次递归：
    撤销选择，path 变回 []。
    继续循环，i = 1：
        选择 2，path 变为 [2]，调用 backtracking(nums, 2)。

10.第五次递归 (startIndex = 2):
    当前路径 [2] 添加到 result。
    选择 3，path 变为 [2, 3]，调用 backtracking(nums, 3)。

11.第六次递归 (startIndex = 3):
    当前路径 [2, 3] 添加到 result，返回。

12.继续回溯：
    撤销选择，path 变回 [2]，返回到上层。

13.继续回溯：
    撤销选择，path 变回 []，返回到第一次递归。

14.继续第一次递归：
    继续循环，i = 2：
        选择 3，path 变为 [3]，调用 backtracking(nums, 3)。

15.第七次递归 (startIndex = 3):
    当前路径 [3] 添加到 result，返回。

16.总结
    递归次数：一共进行了 7 次递归调用。
    撤销选择次数：每次从递归返回时撤销一次选择，因此撤销选择总共 7 次（每次递归后）。
    最终结果：生成的子集为 {}, {1}, {1, 2}, {1, 2, 3}, {1, 3}, {2}, {2, 3}, {3}。
    这样，通过回溯法，程序遍历了所有可能的子集。
*/
```

```c++
//打印幂集
void print(vector<vector<int>>& result) {
    for (const vector<int>& row : result) {
        cout << "{";
        for (const int& a : row) {
            cout << a << (a == row.back() ? "" : ",");
        }
        cout << "}";
        cout << endl;
    }
}
```
