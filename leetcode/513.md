# 513.找树左下角的值
```c++
class Solution {
    // 全局变量
    TreeNode* leftest;

public:
    int findBottomLeftValue(TreeNode* root) {
        queue<TreeNode*> q;
        q.push(root);
        while (!q.empty()) {
            int nodesize = q.size();
            leftest = q.front(); // 记录每层最左边的结点
            for (int i = 1; i <= nodesize; ++i) {
                TreeNode* node = q.front();
                q.pop();
                if (node->left) {
                    q.push(node->left);
                }
                if (node->right) {
                    q.push(node->right);
                }
            }
        }
        return leftest->val;
    }
};
```
