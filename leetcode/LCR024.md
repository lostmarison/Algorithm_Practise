``` c++
//给定单链表的头节点 head ，请反转链表，并返回反转后的链表的头节点。
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        if (head == nullptr) {
            return head;
        }
        ListNode* dummy = new ListNode; // 创建头结点
        dummy->next = nullptr;          // 头结点指针域置空
        ListNode* temp = head;
        while (temp) {
            // 采用头插法即可
            ListNode* s = new ListNode;
            s->val = temp->val;
            s->next = dummy->next;
            dummy->next = s;
            temp = temp->next;
        }
        return dummy->next;
    }
};
```
