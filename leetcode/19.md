# 19.删除链表的倒数第 N 个结点
```c++
class Solution {
public:
    ListNode* removeNthFromEnd(ListNode* head, int n) {
        if (head == nullptr || head->next == nullptr && n == 1) {
            return nullptr;
        }
        // 创建哑结点，否则无法删除头结点
        ListNode* dummy = new ListNode(-1, head);
        ListNode* p = dummy;
        ListNode* q = dummy;
        // q先移动n步
        for (int i = 1; i <= n; ++i) {
            if (q == nullptr) {
                return dummy->next;
            }
            q = q->next;
        }
        // p跟q移动移动直到q到最后一个结点
        while (q->next != nullptr) {
            p = p->next;
            q = q->next;
        }

        // 删除倒数第n个结点
        p->next = p->next->next;

        return dummy->next;
    }
};
```
