# 77.组合
```c++
/*
给定两个整数n和k，返回范围[1, n]中所有可能的k个数的组合。
你可以按任何顺序返回答案。
示例 1：
输入：n = 4, k = 2
输出：
[
  [2,4],
  [3,4],
  [2,3],
  [1,2],
  [1,3],
  [1,4],
]
*/
```

```c++
class Solution {  
private:  
    vector<vector<int>> result;   
    vector<int> temp;   
    /**
     * 回溯函数，用于生成所有可能的组合  
     *
     * @param n 表示可选数字的上限（1 到 n）  
     * @param k 表示每个组合中应包含的数字数量  
     * @param startIndex 表示当前组合路径构建时，可选数字的起始索引  
    void backtracking(int n, int k, int startIndex) {  
        // 如果当前路径的大小等于k，表示找到了一个有效的组合  
        if (path.size() == k) {  
            // 将当前路径添加到结果集中  
            result.push_back(path);  
            return;  
        }  
        // 从startIndex开始遍历到n，尝试将每个数字添加到当前路径中  
        for (int i = startIndex; i <= n; ++i) {  
            // 将当前数字添加到路径中  
            path.push_back(i);  
            // 递归调用backtracking，继续构建路径，但下一次循环时，起始索引为i+1，避免重复  
            backtracking(n, k, i + 1);  
            // 回溯：移除路径中最后一个数字，尝试其他可能的数字  
            path.pop_back();  
        }  
    }  
  
public:  
    vector<vector<int>> combine(int n, int k) {  
        result.clear();  
        path.clear();  
        backtracking(n, k, 1);  
        return result;  
    }  
};
```
