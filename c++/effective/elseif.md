```c++
if (条件1){
    //语句1
}
 
if (条件2){
    //语句2
}else{
    //语句3
}
/*这种格式中，程序会依次判断条件1和条件2是否成立并根据结果决定是否执行语句1和语句2，
也就是说，第一个if语句对第二个if语句没有影响（除非在执行第一个if的时候return）*/

if (条件1) {
    //语句1
}
else if (条件2){
    //语句2
}else{
    //语句3
}
/*if语句和else if语句本质上是互斥的！
也就是说，一旦语句1得到了执行，程序会跳过else if语句，
else if语句中的判断语句以及语句2一定会被跳过；
同时语句2的执行也暗含了条件1判断失败和语句1没有执行；
*/
```
